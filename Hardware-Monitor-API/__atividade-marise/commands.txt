#Comandos MySQL

CREATE DATABASE projeto;

USE projeto;

CREATE TABLE dados ( 
    id int NOT NULL AUTO_INCREMENT, 
    cpu_percent varchar(5) NOT NULL,
    ram_percent varchar(5) NOT NULL, 
    disk_percent varchar(5) NOT NULL, 
    download varchar(8) NOT NULL,
    upload varchar(8) NOT NULL,
    temp_celsius varchar(5) NOT NULL,
    swap_percent varchar(5) NOT NULL,
    tasks varchar(5) NOT NULL,
    clock varchar(8) NOT NULL,
    dataHora datetime NOT NULL,
    PRIMARY KEY (id)
);
